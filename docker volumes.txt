to use in prodction in react (npm run build)

need two images to pull node and nginx

FROM node:current-alpine as builder

WORKDIR /app

COPY package.json .

RUN npm install

COPY . .

RUN npm run build

FROM nginx

COPY --from=builder /app/build /usr/share/nginx/html
